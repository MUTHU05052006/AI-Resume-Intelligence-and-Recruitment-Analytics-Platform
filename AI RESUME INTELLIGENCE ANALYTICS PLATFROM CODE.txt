    def handle_qr_data(self, data):
        try:
            name = data.get("name", "").strip()
            role = data.get("role", "").strip()
            department = data.get("department", "").strip()
            experience = str(data.get("experience", "0")).strip()
            skills = data.get("skills", "").strip()

            # Best option: unique id inside QR
            qr_id = str(data.get("qr_id", "")).strip()

            # Fallback unique key if qr_id is not present
            unique_key = qr_id if qr_id else f"{name}|{role}|{department}|{experience}|{skills}".lower()

            rows = self.read_rows()

            # duplicate check
            for row in rows:
                existing_key = (
                    row.get("SourceFile", "").strip().lower()
                    if row.get("SourceFile", "").strip().startswith("QR::")
                    else f"{row.get('Name','').strip()}|{row.get('Role','').strip()}|"
                         f"{row.get('Department','').strip()}|{row.get('Experience','').strip()}|"
                         f"{row.get('Skills','').strip()}".lower()
                )

                if existing_key == f"qr::{unique_key}" or existing_key == unique_key:
                    self.set_status("Duplicate QR entry ignored.")
                    return

            # fill form
            self.form_widgets["Name"].delete(0, tk.END)
            self.form_widgets["Name"].insert(0, name)

            self.form_widgets["Role"].set(role)

            self.form_widgets["Department"].delete(0, tk.END)
            self.form_widgets["Department"].insert(0, department)

            self.form_widgets["Experience"].delete(0, tk.END)
            self.form_widgets["Experience"].insert(0, experience)

            self.form_widgets["Skills"].delete(0, tk.END)
            self.form_widgets["Skills"].insert(0, skills)

            # optional defaults
            self.form_widgets["Education"].delete(0, tk.END)
            self.form_widgets["Salary"].delete(0, tk.END)
            self.form_widgets["Selected"].set("No")

            self.current_resume_path = f"QR::{unique_key}"
            self.update_score_fields()

            # auto-save to file + MySQL
            record = self.build_record_from_form()
            record["SourceFile"] = f"QR::{unique_key}"

            rows.append(record)
            self.write_rows(rows)
            self.replace_mysql_from_rows(rows)
            self.load_data()
            self.refresh_ui_summary()
            self.clear_form()

            self.set_status("QR data saved automatically.")

        except Exception as e:
            print("QR Error:", e)
            messagebox.showerror("QR Error", str(e))


    def qr_entry_exists(self, unique_key):
        rows = self.read_rows()
        target = f"qr::{unique_key}".lower()

        for row in rows:
            source = row.get("SourceFile", "").strip().lower()
            fallback = (
                f"{row.get('Name','').strip()}|{row.get('Role','').strip()}|"
                f"{row.get('Department','').strip()}|{row.get('Experience','').strip()}|"
                f"{row.get('Skills','').strip()}".lower()
            )

            if source == target or fallback == unique_key.lower():
                return True
        return False

    def replace_mysql_from_rows(self, rows):
        if not self.mysql_available:
            return

        try:
            conn = self.get_mysql_connection()
            cursor = conn.cursor()

            cursor.execute(f"TRUNCATE TABLE {self.table_name}")

            query = f"""
                INSERT INTO {self.table_name}
                (ID, Name, Role, Department, Experience, Education, Skills,
                 Salary, Selected, Score, Status, ResumeScore, MLScore, SourceFile)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
            """

            values = []
            for record in rows:
                values.append((
                    record["ID"], record["Name"], record["Role"], record["Department"],
                    record["Experience"], record["Education"], record["Skills"],
                    record["Salary"], record["Selected"], record["Score"], record["Status"],
                    record["ResumeScore"], record["MLScore"], record["SourceFile"]
                ))

            if values:
                cursor.executemany(query, values)

            conn.commit()
            cursor.close()
            conn.close()

        except Error as e:
            print("MySQL replace error:", e)

    def download_pdf_report(self):
        if not REPORTLAB_AVAILABLE:
            messagebox.showerror(
                "Missing Dependency",
                "reportlab is not installed.\nRun: pip install reportlab"
            )
            return

        rows = self.filtered_rows if self.filtered_rows else self.read_rows()
        if not rows:
            messagebox.showerror("Error", "No candidate data available for PDF report.")
            return

        file_path = filedialog.asksaveasfilename(
            title="Save PDF Report",
            defaultextension=".pdf",
            filetypes=[("PDF files", "*.pdf")]
        )
        if not file_path:
            return

        try:
            total_candidates = len(rows)
            shortlisted = sum(1 for r in rows if r["Status"] == "Shortlisted")
            review = sum(1 for r in rows if r["Status"] == "Review")
            rejected = sum(1 for r in rows if r["Status"] == "Rejected")
            avg_score = round(sum(float(r["Score"] or 0) for r in rows) / total_candidates, 2) if total_candidates else 0
            avg_ml = round(sum(float(r["MLScore"] or 0) for r in rows) / total_candidates, 2) if total_candidates else 0
            avg_resume = round(sum(float(r["ResumeScore"] or 0) for r in rows) / total_candidates, 2) if total_candidates else 0

            top = sorted(rows, key=lambda x: float(x["Score"] or 0), reverse=True)[:5]

            pdf = canvas.Canvas(file_path, pagesize=A4)
            width, height = A4

            y = height - 50
            pdf.setFont("Helvetica-Bold", 16)
            pdf.drawString(40, y, "AI Recruitment Report")

            y -= 25
            pdf.setFont("Helvetica", 10)
            pdf.drawString(40, y, f"Generated on: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")

            y -= 35
            pdf.setFont("Helvetica-Bold", 12)
            pdf.drawString(40, y, "Summary")

            y -= 20
            pdf.setFont("Helvetica", 10)
            summary_lines = [
                f"Total Candidates: {total_candidates}",
                f"Shortlisted: {shortlisted}",
                f"Review: {review}",
                f"Rejected: {rejected}",
                f"Average Final Score: {avg_score}",
                f"Average Resume Score: {avg_resume}",
                f"Average ML Score: {avg_ml}",
            ]

            for line in summary_lines:
                pdf.drawString(50, y, line)
                y -= 16

            y -= 10
            pdf.setFont("Helvetica-Bold", 12)
            pdf.drawString(40, y, "Top 5 Candidates")

            y -= 20
            pdf.setFont("Helvetica", 10)

            for idx, row in enumerate(top, start=1):
                candidate_lines = [
                    f"{idx}. {row['Name']}",
                    f"   Role: {row['Role']} | Department: {row['Department']}",
                    f"   Experience: {row['Experience']} | Education: {row['Education']}",
                    f"   Final Score: {row['Score']} | ResumeScore: {row['ResumeScore']} | MLScore: {row['MLScore']}",
                    f"   Status: {row['Status']}",
                ]

                for line in candidate_lines:
                    if y < 60:
                        pdf.showPage()
                        y = height - 50
                        pdf.setFont("Helvetica", 10)
                    pdf.drawString(50, y, line[:110])
                    y -= 16

                y -= 8

            pdf.save()
            self.set_status(f"PDF report saved: {os.path.basename(file_path)}")
            messagebox.showinfo("Success", f"PDF report downloaded successfully.\n\nSaved to:\n{file_path}")

        except Exception as e:
            messagebox.showerror("Error", f"Failed to generate PDF report.\n\n{e}")

    def ensure_row_schema(self, row):
        fixed = {}
        for col in self.LABELS:
            val = row.get(col, "")
            fixed[col] = val.strip() if isinstance(val, str) else str(val)
        return fixed

    def read_rows(self):
        if not os.path.exists(self.data_file):
            return []
        with open(self.data_file, newline="", encoding="utf-8") as f:
            rows = list(csv.DictReader(f))
        return [self.ensure_row_schema(row) for row in rows]

    def write_rows(self, rows):
        with open(self.data_file, "w", newline="", encoding="utf-8") as f:
            writer = csv.DictWriter(f, fieldnames=self.LABELS)
            writer.writeheader()
            for row in rows:
                writer.writerow(self.ensure_row_schema(row))

    def generate_id(self):
        return "C" + datetime.now().strftime("%Y%m%d%H%M%S%f")[-12:]

    def get_summary_metrics(self):
        rows = self.read_rows()
        total = len(rows)
        shortlisted = sum(1 for r in rows if r["Status"] == "Shortlisted")
        review = sum(1 for r in rows if r["Status"] == "Review")
        rejected = sum(1 for r in rows if r["Status"] == "Rejected")
        avg_score = round(
            sum(float(r["Score"] or 0) for r in rows) / total, 2
        ) if total else 0

        return {
            "Total Candidates": total,
            "Shortlisted": shortlisted,
            "Review": review,
            "Rejected": rejected,
            "Avg Score": avg_score
        }

    def add_education(self, event=None):
        selected = self.edu_combo.get().strip()
        current = self.form_widgets["Education"].get().strip()

        if not selected:
            return

        if current == "":
            new_text = selected
        else:
            existing = [x.strip().lower() for x in current.split(",")]
            if selected.lower() in existing:
                return
            new_text = current + ", " + selected

        self.form_widgets["Education"].delete(0, tk.END)
        self.form_widgets["Education"].insert(0, new_text)
        self.update_score_fields()

    def extract_name_from_resume(self, text, filepath=""):
        lines = [line.strip() for line in (text or "").splitlines() if line.strip()]
        if lines:
            first = lines[0]
            if len(first.split()) <= 5 and not any(ch.isdigit() for ch in first):
                return first.title()
        if filepath:
            return os.path.splitext(os.path.basename(filepath))[0].replace("_", " ").replace("-", " ").title()
        return ""

    def extract_skills(self, text):
        text_lower = str(text).lower()
        found = []
        for skill in self.GLOBAL_SKILLS:
            if skill in text_lower:
                found.append(skill.title())
        return sorted(set(found))

    def infer_education(self, text):
        text_lower = str(text).lower()
        found = []
        for edu in self.EDUCATION_OPTIONS:
            if edu.lower() in text_lower:
                found.append(edu)
        return ", ".join(sorted(set(found)))

    def infer_experience(self, text):
        text_lower = str(text).lower()
        patterns = [r'(\d+)\+?\s+years', r'(\d+)\+?\s+yrs', r'experience\s*[:\-]?\s*(\d+)']
        values = []
        for pattern in patterns:
            matches = re.findall(pattern, text_lower)
            for m in matches:
                try:
                    values.append(int(m))
                except Exception:
                    pass
        return str(max(values)) if values else ""

    def infer_role(self, skills):
        skill_set = {s.lower() for s in skills}
        best_role = "General"
        best_score = -1
        for role, role_skills in self.ROLE_SKILLS.items():
            overlap = len(skill_set.intersection(role_skills))
            if overlap > best_score:
                best_score = overlap
                best_role = role
        return best_role

    def rule_resume_score(self, record):
        score = 0
        role = record["Role"].strip()
        experience = record["Experience"].strip()
        education = record["Education"].strip().lower()
        skills_text = record["Skills"].strip().lower()
        selected = record["Selected"].strip().lower()

        try:
            exp_val = int(experience)
            score += min(exp_val * 5, 30)
        except ValueError:
            pass

        edu_items = [e.strip() for e in education.split(",") if e.strip()]
        for degree in edu_items:
            for key, value in self.EDUCATION_SCORES.items():
                if key in degree:
                    score += value
                    break

        skill_items = [s.strip() for s in skills_text.split(",") if s.strip()]
        skill_set = set(skill_items)

        role_map = self.ROLE_SKILLS.get(role, self.ROLE_SKILLS["General"])
        role_hits = 0
        for skill_name in role_map:
            if skill_name in skill_set:
                score += 8
                role_hits += 1

        score += min(len(skill_items) * 2, 20)
        score += min(role_hits * 3, 12)

        if selected == "yes":
            score += 5

        return min(score, 100)

    def build_feature_text(self, record):
        return " ".join([
            record.get("Role", ""),
            record.get("Department", ""),
            record.get("Experience", ""),
            record.get("Education", ""),
            record.get("Skills", ""),
            record.get("ResumeScore", ""),
        ])

    def ml_predict_score(self, record):
        if self.ml_pipeline is None:
            return 0
        try:
            text = self.build_feature_text(record)
            if hasattr(self.ml_pipeline, "predict_proba"):
                proba = self.ml_pipeline.predict_proba([text])[0][1]
                return int(round(proba * 100))
            pred = self.ml_pipeline.predict([text])[0]
            return 100 if str(pred).lower() in ("1", "yes", "true", "selected") else 0
        except Exception:
            return 0

    def final_score(self, resume_score, ml_score):
        if self.ml_pipeline is not None:
            score = (0.60 * resume_score) + (0.40 * ml_score)
        else:
            score = resume_score
        return int(round(score))

    def calculate_status(self, score):
        if score >= 75:
            return "Shortlisted"
        elif score >= 50:
            return "Review"
        return "Rejected"

    # ---------------- RESUME PARSING ----------------
    def read_resume_file(self, file_path):
        ext = os.path.splitext(file_path)[1].lower()
        try:
            if ext == ".txt":
                with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
                    return f.read()

            if ext == ".pdf":
                if pdfplumber is None:
                    raise RuntimeError("pdfplumber is not installed. Run: pip install pdfplumber")
                text_parts = []
                with pdfplumber.open(file_path) as pdf:
                    for page in pdf.pages:
                        text_parts.append(page.extract_text() or "")
                return "\n".join(text_parts)

            if ext == ".docx":
                if Document is None:
                    raise RuntimeError("python-docx is not installed. Run: pip install python-docx")
                doc = Document(file_path)
                return "\n".join([p.text for p in doc.paragraphs])

            raise RuntimeError("Unsupported file format. Please upload TXT, PDF, or DOCX.")
        except Exception as e:
            raise RuntimeError(f"Could not read resume file.\n{e}")

    # ---------------- ML MODEL ----------------
    def load_model_if_exists(self):
        if not os.path.exists(self.model_file):
            self.model_info_var.set("ML Model: Not trained")
            return

        try:
            if joblib is not None:
                self.ml_pipeline = joblib.load(self.model_file)
            else:
                with open(self.model_file, "rb") as f:
                    self.ml_pipeline = pickle.load(f)
            self.model_info_var.set("ML Model: Trained")
        except Exception:
            self.ml_pipeline = None
            self.model_info_var.set("ML Model: Failed to load")

    def train_model_from_dataset(self):
        if not SKLEARN_AVAILABLE:
            messagebox.showerror("Missing Dependency", "scikit-learn is not installed.\nRun: pip install scikit-learn")
            return

        rows = self.read_rows()
        if len(rows) < 6:
            messagebox.showerror("Error", "Need at least 6 records to train a model.")
            return

        usable = [row for row in rows if row.get("Selected", "").strip().lower() in ("yes", "no")]
        if len(usable) < 6:
            messagebox.showerror("Error", "Not enough labeled records with Selected = Yes/No.")
            return

        texts = []
        labels = []
        for row in usable:
            texts.append(self.build_feature_text(row))
            labels.append(1 if row["Selected"].strip().lower() == "yes" else 0)

        try:
            if len(set(labels)) < 2:
                messagebox.showerror("Error", "Training requires both Yes and No labels.")
                return

            X_train, X_test, y_train, y_test = train_test_split(
                texts, labels, test_size=0.3, random_state=42, stratify=labels
            )

            pipeline = Pipeline([
                ("tfidf", TfidfVectorizer(stop_words="english")),
                ("clf", LogisticRegression(max_iter=1000))
            ])
            pipeline.fit(X_train, y_train)

            _ = pipeline.predict(X_test)

            self.ml_pipeline = pipeline

            if joblib is not None:
                joblib.dump(self.ml_pipeline, self.model_file)
            else:
                with open(self.model_file, "wb") as f:
                    pickle.dump(self.ml_pipeline, f)

            self.model_info_var.set("ML Model: Trained")
            self.recalculate_all_scores(silent=True)
            self.set_status("ML model trained successfully.")

        except Exception as e:
            messagebox.showerror("Error", f"Model training failed.\n\n{e}")

    def recalculate_all_scores(self, silent=False):
        rows = self.read_rows()
        if not rows:
            if not silent:
                messagebox.showerror("Error", "No records found.")
            return

        updated_rows = []

        for row in rows:
            row = self.ensure_row_schema(row)

            resume_score = self.rule_resume_score(row)
            ml_score = self.ml_predict_score({
                **row,
                "ResumeScore": str(resume_score)
            })
            total = self.final_score(resume_score, ml_score)

            row["ResumeScore"] = str(resume_score)
            row["MLScore"] = str(ml_score)
            row["Score"] = str(total)
            row["Status"] = self.calculate_status(total)

            updated_rows.append(row)

        self.write_rows(updated_rows)
        self.replace_mysql_from_rows(updated_rows)
        self.load_data()
        self.refresh_ui_summary()
        self.update_score_fields()

        if not silent:
            self.set_status("All candidate scores recalculated.")

        if not rows:

        def sort_key(row):

    def import_csv(self):
        file_path = filedialog.askopenfilename(filetypes=[("CSV files", "*.csv")], title="Select CSV File")
        if not file_path:
            return

        try:
            with open(file_path, newline="", encoding="utf-8") as f:
                reader = csv.DictReader(f)
                imported_rows = list(reader)

            if not imported_rows:
                messagebox.showerror("Error", "Selected CSV is empty.")
                return

            final_rows = []
            for row in imported_rows:
                role = row.get("Role", "General").strip() or "General"
                record = {
                    "ID": row.get("ID", "").strip() or self.generate_id(),
                    "Name": row.get("Name", "").strip(),
                    "Role": role,
                    "Department": row.get("Department", "").strip(),
                    "Experience": row.get("Experience", "").strip(),
                    "Education": row.get("Education", "").strip(),
                    "Skills": row.get("Skills", "").strip(),
                    "Salary": row.get("Salary", "").strip(),
                    "Selected": self.normalize_yes_no(row.get("Selected", "No")),
                    "ResumeScore": "",
                    "MLScore": "",
                    "Score": "",
                    "Status": "",
                    "SourceFile": row.get("SourceFile", "").strip()
                }

                resume_score = self.rule_resume_score(record)
                ml_score = self.ml_predict_score({**record, "ResumeScore": str(resume_score)})
                total = self.final_score(resume_score, ml_score)

                record["ResumeScore"] = str(resume_score)
                record["MLScore"] = str(ml_score)
                record["Score"] = str(total)
                record["Status"] = self.calculate_status(total)
                final_rows.append(record)

            self.write_rows(final_rows)
            self.replace_mysql_from_rows(final_rows)
            self.load_data()
            self.refresh_ui_summary()
            self.clear_form()
            self.set_status("CSV imported successfully into shared data file.")

        except Exception as e:
            messagebox.showerror("Error", f"Failed to import CSV.\n\n{e}")
